# For damaging a province
province_event = {
	id = mrd.2
	hidden = yes

	trigger = {
		devastation = 10
		OR = {
			base_tax = 2
			base_production = 2
			base_manpower = 2
		}
	}

	mean_time_to_happen = {
		months = 480

		# Devastation

		modifier = {
			factor = 0.9
			devastation = 20
		}

		modifier = {
			factor = 0.9
			devastation = 30
		}

		modifier = {
			factor = 0.9
			devastation = 40
		}

		modifier = {
			factor = 0.9
			devastation = 50
		}

		modifier = {
			factor = 0.9
			devastation = 60
		}

		modifier = {
			factor = 0.9
			devastation = 70
		}

		modifier = {
			factor = 0.9
			devastation = 80
		}

		modifier = {
			factor = 0.9
			devastation = 90
		}
	}

	option = {
		if = {
			limit = { has_enough_all = yes }
			random_list = {
				33 = { remove_tax = yes }
				33 = { remove_production = yes }
				33 = { remove_manpower = yes }
			}
		}
		else_if = {
			limit = { has_enough_tax_production = yes }
			random_list = {
				33 = { remove_tax = yes }
				33 = { remove_production = yes }
			}
		}
		else_if = {
			limit = { has_enough_tax_manpower = yes }
			random_list = {
				33 = { remove_tax = yes }
				33 = { remove_manpower = yes }
			}
		}
		else_if = {
			limit = { has_enough_production_manpower = yes }
			random_list = {
				33 = { remove_production = yes }
				33 = { remove_manpower = yes }
			}
		}
		else_if = {
			limit = { has_enough_tax = yes }
			remove_tax = yes
		}
		else_if = {
			limit = { has_enough_production = yes }
			remove_production = yes
		}
		else = { remove_manpower = yes }

		if = {
			limit = {
				owner = {
					OR = {
						is_vassal = yes
						is_lesser_in_union = yes
					}
				}
			}
			owner = {
				overlord = {
					subtract_variable = {
						which = grossVassalDevelopment
						value = 1
					}
				}
			}
		}

		if = {
			limit = { owner = { is_colonial_nation = yes } }
			owner = {
				colonial_parent = {
					subtract_variable = {
						which = grossVassalDevelopment
						value = 1
					}
				}
			}
		}

		owner = {
			subtract_variable = {
				which = grossDevelopment
				value = 1
			}
		}
	}
}
